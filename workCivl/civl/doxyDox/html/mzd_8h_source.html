257  if (__M4RI_UNLIKELY(n == 0)) {

455 static inline void mzd_col_swap_in_rows(mzd_t *M, rci_t const cola, rci_t const colb, rci_t const start_row, rci_t const stop_row) {
456  if (cola == colb)
457  return;
458 
459  rci_t const _cola = cola;
460  rci_t const _colb = colb;
461 
462  wi_t const a_word = _cola / m4ri_radix;
463  wi_t const b_word = _colb / m4ri_radix;
464 
465  int const a_bit = _cola % m4ri_radix;
466  int const b_bit = _colb % m4ri_radix;
467 
468  word* RESTRICT ptr = mzd_row(M, start_row);
469  int max_bit = MAX(a_bit, b_bit);
470  int count_remaining = stop_row - start_row;
471  int min_bit = a_bit + b_bit - max_bit;
472  int block = mzd_row_to_block(M, start_row);
473  int offset = max_bit - min_bit;
474  word mask = m4ri_one << min_bit;
475  int count = MIN(mzd_remaining_rows_in_block(M, start_row), count_remaining);
476 
477  // Apparently we're calling with start_row == stop_row sometimes (seems a bug to me).
478  if (count <= 0)
479  return;
480 
481  if (a_word == b_word) {
482  while(1) {
483  count_remaining -= count;
484  ptr += a_word;
485  int fast_count = count / 4;
486  int rest_count = count - 4 * fast_count;
487  word xor_v[4];
488  wi_t const rowstride = M->rowstride;
489  while (fast_count--) {
490  xor_v[0] = ptr[0];
491  xor_v[1] = ptr[rowstride];
492  xor_v[2] = ptr[2 * rowstride];
493  xor_v[3] = ptr[3 * rowstride];
494  xor_v[0] ^= xor_v[0] >> offset;
495  xor_v[1] ^= xor_v[1] >> offset;
496  xor_v[2] ^= xor_v[2] >> offset;
497  xor_v[3] ^= xor_v[3] >> offset;
498  xor_v[0] &= mask;
499  xor_v[1] &= mask;
500  xor_v[2] &= mask;
501  xor_v[3] &= mask;
502  xor_v[0] |= xor_v[0] << offset;
503  xor_v[1] |= xor_v[1] << offset;
504  xor_v[2] |= xor_v[2] << offset;
505  xor_v[3] |= xor_v[3] << offset;
506  ptr[0] ^= xor_v[0];
507  ptr[rowstride] ^= xor_v[1];
508  ptr[2 * rowstride] ^= xor_v[2];
509  ptr[3 * rowstride] ^= xor_v[3];
510  ptr += 4 * rowstride;
511  }
512  while (rest_count--) {
513  word xor_v = *ptr;
514  xor_v ^= xor_v >> offset;
515  xor_v &= mask;
516  *ptr ^= xor_v | (xor_v << offset);
517  ptr += rowstride;
518  }
519  block++;
520  if ((count = MIN(mzd_rows_in_block(M, block), count_remaining)) <= 0)
521  break;
522  ptr = mzd_first_row_next_block(M, block);
523  }
524  } else {
525  word* RESTRICT min_ptr;
526  wi_t max_offset;
527  if (min_bit == a_bit) {
528  min_ptr = ptr + a_word;
529  max_offset = b_word - a_word;
530  } else {
531  min_ptr = ptr + b_word;
532  max_offset = a_word - b_word;
533  }
534  while(1) {
535  count_remaining -= count;
536  wi_t const rowstride = M->rowstride;
537  while(count--) {
538  word xor_v = (min_ptr[0] ^ (min_ptr[max_offset] >> offset)) & mask;
539  min_ptr[0] ^= xor_v;
540  min_ptr[max_offset] ^= xor_v << offset;
541  min_ptr += rowstride;
542  }
543  block++;
544  if ((count = MIN(mzd_rows_in_block(M,+block), count_remaining)) <= 0)
545  break;
546  ptr = mzd_first_row_next_block(M, block);
547  if (min_bit == a_bit)
548  min_ptr = ptr + a_word;
549  else
550  min_ptr = ptr + b_word;
551  }
552  }
553 
554  __M4RI_DD_MZD(M);
555 }

662 static inline void mzd_row_add_offset(mzd_t *M, rci_t dstrow, rci_t srcrow, rci_t coloffset) {
663  assert(dstrow < M->nrows && srcrow < M->nrows && coloffset < M->ncols);
664  wi_t const startblock= coloffset/m4ri_radix;
665  wi_t wide = M->width - startblock;
666  word *src = M->rows[srcrow] + startblock;
667  word *dst = M->rows[dstrow] + startblock;
668  word const mask_begin = __M4RI_RIGHT_BITMASK(m4ri_radix - coloffset % m4ri_radix);
669  word const mask_end = M->high_bitmask;
670 
671  *dst++ ^= *src++ & mask_begin;
672  --wide;
673 
674 #if __M4RI_HAVE_SSE2
675  int not_aligned = __M4RI_ALIGNMENT(src,16) != 0; /* 0: Aligned, 1: Not aligned */
676  if (wide > not_aligned + 1) /* Speed up for small matrices */
677  {
678  if (not_aligned) {
679  *dst++ ^= *src++;
680  --wide;
681  }
682  /* Now wide > 1 */
683  __m128i* __src = (__m128i*)src;
684  __m128i* __dst = (__m128i*)dst;
685  __m128i* const eof = (__m128i*)((unsigned long)(src + wide) & ~0xFUL);
686  do
687  {
688  __m128i xmm1 = _mm_xor_si128(*__dst, *__src);
689  *__dst++ = xmm1;
690  }
691  while(++__src < eof);
692  src = (word*)__src;
693  dst = (word*)__dst;
694  wide = ((sizeof(word)*wide)%16)/sizeof(word);
695  }
696 #endif
697  wi_t i = -1;
698  while(++i < wide)
699  dst[i] ^= src[i];
700  /*
701  * Revert possibly non-zero excess bits.
702  * Note that i == wide here, and wide can be 0.
703  * But really, src[wide - 1] is M->rows[srcrow][M->width - 1] ;)
704  * We use i - 1 here to let the compiler know these are the same addresses
705  * that we last accessed, in the previous loop.
706  */
707  dst[i - 1] ^= src[i - 1] & ~mask_end;
708 
709  __M4RI_DD_ROW(M, dstrow);
710 }
